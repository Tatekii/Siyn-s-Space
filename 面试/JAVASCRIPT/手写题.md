```javascript
const isObject = (obj) => typeof obj === 'object' && obj !== null;

function deepClone(source,cache = new WeakMap()){
	if(!isObject(obj) || cache.has(source)){
		return source
	}

	const constructor = source.constructor
	const res = constructor === Object ? {} : new constructor(source)

	cache.set(source,res)

	Reflect.ownKeys(source).forEach(k=>{
		res[k] = deepClone(source[k],cache)
	})
	
	return res
}
```
